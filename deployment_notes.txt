#环境部署流程：

##1. 安装好conda
wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-aarch64.sh -O ~/miniconda.sh
bash ~/miniconda.sh -b -p ~/miniconda
~/miniconda/bin/conda init bash
source ~/.bashrc

##2. 新建虚拟环境
conda create -n sigprocess python==3.10
pip install -r requirement.txt

##3. 安装cuda相关环境

sudo apt install -y libcurand-12-6
sudo apt install -y cuda-nvrtc-12-6
sudo apt install -y cuda-cudart-dev-12-6
sudo apt install -y libcufft-12-6
sudo apt install -y libcublas-12-6 libcublas-dev-12-6 libcufft-12-6 libcufft-dev-12-6 libcurand-12-6 libcurand-dev-12-6 libcusolver-12-6 libcusolver-dev-12-6 libcusparse-12-6 libcusparse-dev-12-6
